        p = self.table[index]

        if self.table[index].value is None:

        p = self.table[index]
        if p.key == key:
            self.table[index] = ListNode() if p.next is None else p.next
            return

        # ì—°ê²° ë¦¬ìŠ¤íŠ¸ ë…¸ë“œ ì‚­ì œ
        prev = p
        while p:
            if p.key == key:
                prev.next = p.next
                return
            prev, p = p, p.next
